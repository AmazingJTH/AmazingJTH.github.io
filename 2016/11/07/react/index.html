react

简介

  1. 本质：如何将来自服务器端或者用户输入的动态数据高效的反映到复杂的用户界面上
    用于开发数据不断变化的程序，实现前端界面的高效率高性能开发
  2. 虚拟DOM，数据变化时重新构建DOM树，将当前DOM树和之前的进行对比，得到结构区别，将需要变化的部分进行更新
    虚拟DOM是内存数据，性能高实际DOM操作仅仅是diff部分

新建comment-list.js 在comment-box.js中引入

'use strict';
import React from 'react';
import CommentItem from './comment-item.js';
class CommentList extends React.Component {
  render(){
    let CommentNodes = this.props.data.map(comment,index => {
      // console.log(comment);
      return{
        <CommentItem key={index} author={comment.author} date={comment.date}>
          {comment.content}
        </CommentItem>
      }
    })
    return(
      <div>
        { CommentNodes }
      </div>
    )
  }
}
export {CommentList as default};

constructor(props){
  super(props)
  console.log(this.props.url);
  //定义当前组件的状态
  this.state = {data:[]};
  fetch(this.props.url).then(response => response.json()).then(data =>{
    // console.log(data);
    this.setState({data:data});
  }).catch(e =>{
    console.log("Oops, error");
  })
}
render(
  <CommentList data = {this.state.data}>
